 probaClasses <- function(data){
    dataframe <- as.data.frame(table(data));
    nbMoins <- dataframe[1,2];
    nbPlus <- dataframe[2,2];
    total <- nbPlus + nbMoins;
    freqMoins <- nbMoins/total;
    freqPlus <- nbPlus/total;
    truc <- c("-"=freqMoins, "+"=freqPlus);
    return(truc);
}
